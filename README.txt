# Bash Script for countdown

# It gives you countdown of your upcoming or past spent days
# It can be anything from birthday, holiday or even how long you've lived in Earth. 

# To Install 
# sudo ./install-countdown.sh

# To use the app

# countdown [date]		# countdown 10Jan
# countdown ddmonthyear 	# countdown 10Jan2001
# countdown monthddyear		# countdown Jan102001
# countdown monthdd		# countdown Jan10
# countdown "month dd year"	# countdown "Jan 10 2001"
